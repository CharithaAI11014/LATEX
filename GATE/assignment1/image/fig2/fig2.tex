    % Define coordinates for the points
    \coordinate (A) at (0,0);

    \coordinate (D) at (0,2);

    % Internal horizontal force between rectangles

    \draw[->] (B) -- ++(0,-1.5) node[midway, right] {$q_1$};  % Bottom right unknown force (q1)
